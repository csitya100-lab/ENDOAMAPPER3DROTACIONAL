    <script type="module">
        import * as THREE from 'three';
        import { OrbitControls } from 'three/addons/controls/OrbitControls.js';

        // --- GLOBAL VARIABLES & STATE ---
        let canvas, renderer, scene;
        let anatomyGroup, lesionGroup;
        const views = []; 
        let currentSeverity = 'superficial';
        const raycaster = new THREE.Raycaster();
        const mouse = new THREE.Vector2();
        
        const COLORS = {
            superficial: 0xef4444, // Red
            moderate: 0xf97316,   // Orange
            deep: 0x3b82f6        // Blue
        };

        const statusEl = document.getElementById('debug-status');

        // --- EXPOSE FUNCTIONS TO WINDOW FOR HTML BUTTONS ---
        window.setSeverity = (sev) => {
            currentSeverity = sev;
            // Update UI
            document.querySelectorAll('.severity-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(`btn-${sev}`).classList.add('active');
        };

        window.addTestLesion = () => {
            // Add a lesion at a random position on the uterus body
            // Icosahedron radius is approx 1.5, slightly scaled
            const theta = Math.random() * Math.PI * 2;
            const phi = Math.random() * Math.PI;
            const r = 1.5; // Approx radius
            
            const x = r * Math.sin(phi) * Math.cos(theta);
            const y = (r * Math.sin(phi) * Math.sin(theta)) + 0.5; // +0.5 is uterus y offset
            const z = r * Math.cos(phi);
            
            console.log(`Adding Test Lesion at: ${x.toFixed(2)}, ${y.toFixed(2)}, ${z.toFixed(2)}`);
            createLesionMesh(new THREE.Vector3(x, y, z), currentSeverity);
        };

        window.clearLesions = () => {
            if (!lesionGroup) return;
            // Clear all children
            while(lesionGroup.children.length > 0){ 
                lesionGroup.remove(lesionGroup.children[0]); 
            }
            console.log('All lesions cleared');
            updateStatus();
        };

        function updateStatus() {
            if (statusEl && lesionGroup) {
                statusEl.innerText = `Lesions: ${lesionGroup.children.length}`;
            }
        }

        // --- INITIALIZATION ---
        function init() {
            canvas = document.getElementById('c');
            renderer = new THREE.WebGLRenderer({ canvas: canvas, antialias: true, alpha: true });
            renderer.setPixelRatio(window.devicePixelRatio);
            renderer.shadowMap.enabled = true;
            renderer.setScissorTest(true);

            scene = new THREE.Scene();
            
            // LIGHTS
            const ambientLight = new THREE.AmbientLight(0xffffff, 0.4);
            scene.add(ambientLight);

            const dirLight = new THREE.DirectionalLight(0xffffff, 1.0);
            dirLight.position.set(5, 10, 7);
            dirLight.castShadow = true;
            dirLight.shadow.mapSize.width = 1024;
            dirLight.shadow.mapSize.height = 1024;
            scene.add(dirLight);

            const backLight = new THREE.DirectionalLight(0xaaaaff, 0.4);
            backLight.position.set(-5, 2, -10);
            scene.add(backLight);

            // ANATOMY GROUP (Targets for clicking)
            anatomyGroup = new THREE.Group();
            
            // Materials
            const organMaterial = new THREE.MeshPhysicalMaterial({ color: 0xe8a0a0, roughness: 0.3, metalness: 0.1, clearcoat: 0.3 });
            const cervixMaterial = new THREE.MeshPhysicalMaterial({ color: 0xd48888, roughness: 0.4, metalness: 0.0 });
            const ovaryMaterial = new THREE.MeshPhysicalMaterial({ color: 0xd4c5a0, roughness: 0.5, metalness: 0.0 });

            // 1. Uterus Body
            const bodyGeo = new THREE.IcosahedronGeometry(1.5, 2);
            const uterusBody = new THREE.Mesh(bodyGeo, organMaterial);
            uterusBody.position.set(0, 0.5, 0);
            uterusBody.scale.set(1, 1.1, 0.9); 
            uterusBody.castShadow = true;
            uterusBody.receiveShadow = true;
            anatomyGroup.add(uterusBody);

            // 2. Cervix
            const cervixGeo = new THREE.CylinderGeometry(0.7, 0.7, 1.5, 32);
            const cervix = new THREE.Mesh(cervixGeo, cervixMaterial);
            cervix.position.set(0, -1.2, 0);
            cervix.castShadow = true;
            cervix.receiveShadow = true;
            anatomyGroup.add(cervix);

            // 3. Vagina
            const vaginaGeo = new THREE.CylinderGeometry(0.9, 0.8, 1.2, 32, 1, true);
            const vagina = new THREE.Mesh(vaginaGeo, cervixMaterial);
            vagina.position.set(0, -2.5, 0);
            vagina.material = cervixMaterial.clone();
            vagina.material.side = THREE.DoubleSide;
            vagina.receiveShadow = true;
            anatomyGroup.add(vagina);

            // 4. Tubes
            function createTube(isRight) {
                const xMult = isRight ? 1 : -1;
                const curve = new THREE.CatmullRomCurve3([
                    new THREE.Vector3(1.2 * xMult, 1.5, 0),
                    new THREE.Vector3(2.0 * xMult, 1.8, 0.2),
                    new THREE.Vector3(2.8 * xMult, 1.2, 0.5),
                    new THREE.Vector3(3.0 * xMult, 0.5, 0),
                ]);
                const tube = new THREE.Mesh(new THREE.TubeGeometry(curve, 32, 0.15, 8, false), organMaterial);
                tube.castShadow = true;
                tube.receiveShadow = true;
                return tube;
            }
            anatomyGroup.add(createTube(true));
            anatomyGroup.add(createTube(false));

            // 5. Ovaries
            const ovaryGeo = new THREE.SphereGeometry(0.6, 32, 32);
            const rightOvary = new THREE.Mesh(ovaryGeo, ovaryMaterial);
            rightOvary.position.set(3.0, 0.2, 0);
            rightOvary.castShadow = true;
            anatomyGroup.add(rightOvary);

            const leftOvary = new THREE.Mesh(ovaryGeo, ovaryMaterial);
            leftOvary.position.set(-3.0, 0.2, 0);
            leftOvary.castShadow = true;
            anatomyGroup.add(leftOvary);

            scene.add(anatomyGroup);

            // LESION GROUP
            lesionGroup = new THREE.Group();
            scene.add(lesionGroup);

            // VIEWS CONFIG
            function createView(elementId, cameraType, camPos, upVec, bgColor) {
                const container = document.getElementById(elementId);
                let camera, controls;

                if (cameraType === 'perspective') {
                    camera = new THREE.PerspectiveCamera(50, 1, 0.1, 100);
                    camera.position.set(...camPos);
                    controls = new OrbitControls(camera, container);
                    controls.enableDamping = true;
                    controls.dampingFactor = 0.05;
                    controls.enableZoom = true;
                } else {
                    const fr = 5;
                    camera = new THREE.OrthographicCamera(-fr, fr, fr, -fr, 0.1, 100);
                    camera.position.set(...camPos);
                    camera.zoom = 1.2;
                    if (upVec) camera.up.set(...upVec);
                    camera.lookAt(0, 0, 0);
                    controls = new OrbitControls(camera, container);
                    controls.enableRotate = false;
                    controls.enableZoom = true;
                    controls.enablePan = true;
                }

                const view = { container, camera, controls, background: new THREE.Color(bgColor) };
                
                // ATTACH CLICK LISTENER FOR RAYCASTING
                // We use pointer down/up to distinguish between click and drag
                container.addEventListener('pointerdown', (event) => onPointerDown(event, view));
                
                return view;
            }

            views.push(createView('view-main', 'perspective', [5, 5, 8], null, 0x1a1b26));
            views.push(createView('view-sagittal', 'ortho', [20, 0, 0], [0, 1, 0], 0x13141f));
            views.push(createView('view-coronal', 'ortho', [0, 0, 20], [0, 1, 0], 0x13141f));
            views.push(createView('view-posterior', 'ortho', [0, 0, -20], [0, 1, 0], 0x13141f));

            window.addEventListener('resize', updateSize);
            updateSize();
            animate();
        }

        // --- LESION CREATION ---
        function createLesionMesh(position, severity) {
            const geometry = new THREE.SphereGeometry(0.15, 16, 16);
            const material = new THREE.MeshStandardMaterial({
                color: COLORS[severity],
                emissive: COLORS[severity],
                emissiveIntensity: 0.5,
                roughness: 0.2
            });
            const mesh = new THREE.Mesh(geometry, material);
            mesh.position.copy(position);
            lesionGroup.add(mesh);
            updateStatus();
        }

        // --- INTERACTION ---
        let dragStart = { x: 0, y: 0 };
        
        function onPointerDown(event, view) {
            dragStart.x = event.clientX;
            dragStart.y = event.clientY;
            
            // We listen for pointer up on the specific container to finalize click
            // This prevents adding lesions when just dragging to rotate
            const onPointerUp = (e) => {
                const dist = Math.sqrt(Math.pow(e.clientX - dragStart.x, 2) + Math.pow(e.clientY - dragStart.y, 2));
                if (dist < 5) { // 5px threshold for "click"
                    handleCanvasClick(e, view);
                }
                view.container.removeEventListener('pointerup', onPointerUp);
            };
            view.container.addEventListener('pointerup', onPointerUp);
        }

        function handleCanvasClick(event, view) {
            // Get click coordinates relative to the view container
            const rect = view.container.getBoundingClientRect();
            
            // Calculate mouse position in normalized device coordinates (-1 to +1) for the specific view
            mouse.x = ((event.clientX - rect.left) / rect.width) * 2 - 1;
            mouse.y = -((event.clientY - rect.top) / rect.height) * 2 + 1;

            // Update the picking ray with the camera and mouse position
            raycaster.setFromCamera(mouse, view.camera);

            // Calculate objects intersecting the picking ray
            // We interact with anatomyGroup children recursively
            const intersects = raycaster.intersectObjects(anatomyGroup.children, true);

            if (intersects.length > 0) {
                const point = intersects[0].point;
                console.log(`[3D Click] Lesion created at: X=${point.x.toFixed(3)}, Y=${point.y.toFixed(3)}, Z=${point.z.toFixed(3)} | Severity: ${currentSeverity}`);
                
                createLesionMesh(point, currentSeverity);
            }
        }

        function updateSize() {
            if (canvas && canvas.parentElement) {
                const width = canvas.parentElement.clientWidth;
                const height = canvas.parentElement.clientHeight;
                if (canvas.width !== width || canvas.height !== height) {
                    renderer.setSize(width, height, false);
                }
            }
        }

        function animate() {
            requestAnimationFrame(animate);
            updateSize();

            views.forEach(view => {
                if (view.controls) view.controls.update();
            });

            const canvasRect = canvas.getBoundingClientRect();

            views.forEach((view) => {
                const rect = view.container.getBoundingClientRect();
                
                // Calculate position relative to canvas (which is same as window usually in this layout)
                // However, since canvas matches container, we use relative math
                const relLeft = rect.left - canvasRect.left;
                
                // Scissor Y is from bottom
                const bottom = canvasRect.height - (relLeft < 0 ? 0 : (rect.top - canvasRect.top)) - rect.height;

                renderer.setViewport(relLeft, bottom, rect.width, rect.height);
                renderer.setScissor(relLeft, bottom, rect.width, rect.height);
                
                renderer.setClearColor(view.background);
                renderer.clear();
                renderer.render(scene, view.camera);
            });
        }

        init();
    </script>
